// ============================================================================= // VERIXIA SEO ENGINE WRAPPER — OPEN MODE (v2: live-data enriched) // Wraps tokenriskseoengine.js to: // 1. Hit Jupiter endpoints DIRECTLY for rich live data (price, 24h, volume, // liquidity, market cap), selecting the HIGHEST-LIQUIDITY token match so // ambiguous symbols (e.g. "bonk") resolve to the real asset + its mint. // 2. Forward the engine's already-fetched payload.observations into the // expansion prompt so no live data the engine computed is wasted. // 3. Run TWO expansion passes IN PARALLEL, score each with the engine's own // validateSeoContent, and keep the HIGHER-SCORING one. Always return one // result, even if both score below threshold. No throwing, no fallbacks. // 4. Bypass the engine's strict validator — accept almost anything that has // real content (no lists, no HTML, no disclaimers). // 5. Add framingName to the payload. // Does NOT edit tokenriskseoengine.js at all. // ============================================================================= import fetch from "node-fetch"; import { generateSeoPagePayload as __origGenerateSeoPagePayload, detectSeoIntent, detectSubIntent, extractKeywordSubject, validateSeoContent, fetchJupQuote, fetchJupPrices, fetchJupRecent, fetchJupTokenSearch, resetOutputGate, pageJitter, selectFramingForKeyword, getBuildReport, resetAntiRepetition, } from "./tokenriskseoengine.js"; // ----------------------------------------------------------------------------- // Config // ----------------------------------------------------------------------------- const TARGET_MIN_WORDS = 500; const OPENAI_MODEL = process.env.OPENAI_SEO_MODEL || "gpt-4.1-mini"; const OPENAI_API_URL = process.env.OPENAI_SEO_API_URL || "https://api.openai.com/v1/chat/completions"; const EXPANSION_TIMEOUT_MS = 60_000; const EXPANSION_MAX_TOKENS = 2200; // Two parallel expansion passes. Different temps so the two candidates diverge. const EXPANSION_TEMPERATURES = [0.55, 0.85]; // Direct Jupiter config (mirrors the big file; wrapper owns its own calls). const JUP_BASE = process.env.JUP_SEO_BASE || "https://lite-api.jup.ag"; const JUP_TOKENS_SEARCH = `${JUP_BASE}/tokens/v2/search`; const JUP_TIMEOUT_MS = 6000; const JUP_CACHE_TTL = Number(process.env.JUP_SEO_CACHE_TTL_MS || 60_000); const JUP_CACHE_MAX = Number(process.env.JUP_SEO_CACHE_MAX || 2000); // bound memory across a large build // ----------------------------------------------------------------------------- // Helpers // ----------------------------------------------------------------------------- function wordCount(s) { return String(s || "").trim().split(/\s+/).filter(Boolean).length; } function splitParagraphs(text) { return String(text || "") .replace(/\r/g, "") .split(/\n\s*\n/) .map(p => p.trim()) .filter(Boolean); } function looksLikeListOrHtml(text) { const s = String(text || ""); if (/^[-*#]/m.test(s)) return true; if (/<[a-z][\s\S]*?>/i.test(s)) return true; return false; } function fetchWithTimeout(url, opts, timeoutMs) { const controller = new AbortController(); const t = setTimeout(() => controller.abort(), timeoutMs); return fetch(url, { ...opts, signal: controller.signal }).finally(() => clearTimeout(t)); } // Format a USD figure compactly: $1.2M, $940K, $0.00001234. function fmtUsd(n) { const v = Number(n); if (!isFinite(v)) return null; if (v >= 1_000_000_000) return `$${(v / 1_000_000_000).toFixed(2)}B`; if (v >= 1_000_000) return `$${(v / 1_000_000).toFixed(2)}M`; if (v >= 1_000) return `$${(v / 1_000).toFixed(1)}K`; if (v >= 1) return `$${v.toFixed(2)}`; if (v > 0) return `$${v.toPrecision(4)}`; return null; } // Plain-decimal price, no exponential — an LLM reproduces "$0.0000123" verbatim // but rarely "$1.23e-5", and the weave scorer matches on the exact string. function fmtPriceUsd(n) { const v = Number(n); if (!isFinite(v) || v <= 0) return null; if (v >= 1) return `$${v.toFixed(2)}`; if (v >= 0.01) return `$${v.toFixed(4)}`; if (v >= 1e-9) { // 4 significant figures as a plain decimal, trailing zeros stripped. let s = v.toFixed(12); s = s.replace(/0+$/, ""); // keep at most 4 significant digits after the leading zeros const m = s.match(/^0\.0*(\d{1,4})/); if (m) { const lead = s.slice(0, s.indexOf(m[1])); s = lead + m[1]; } return `$${s}`; } return null; // sub-nano price: not meaningfully reproducible, skip the figure } function fmtPct(n) { const v = Number(n); if (!isFinite(v)) return null; return `${v > 0 ? "+" : ""}${v.toFixed(2)}%`; } // Extract metrics from a Jupiter /tokens/v2/search entry. // Schema confirmed against Jupiter dev docs (dev.jup.ag/api-reference/tokens/v2/search): // top-level: id, name, symbol, usdPrice, mcap, fdv, liquidity, holderCount, // organicScore (0-100 float), organicScoreLabel, isVerified // nested stats blocks: stats5m / stats1h / stats6h / stats24h, each with // priceChange, buyVolume, sellVolume, numTraders, ... // There is NO top-level volume24h or priceChange24h — volume is buyVolume + // sellVolume inside the chosen stats block; 24h change is stats24h.priceChange. function statsVolume(stats) { if (!stats) return null; const buy = Number(stats.buyVolume); const sell = Number(stats.sellVolume); const haveBuy = isFinite(buy); const haveSell = isFinite(sell); if (!haveBuy && !haveSell) return null; return (haveBuy ? buy : 0) + (haveSell ? sell : 0); } function readTokenMetrics(t) { if (!t || typeof t !== "object") return { mint: "" }; const liquidity = Number(t.liquidity); // Prefer 24h volume; fall back to 6h then 1h so newer tokens still show data. const volume24h = statsVolume(t.stats24h) ?? statsVolume(t.stats6h) ?? statsVolume(t.stats1h); // mcap can be null on fresh tokens; fdv is the documented fallback. const mcap = Number(t.mcap ?? t.fdv); const price = Number(t.usdPrice); // 24h price change lives in the stats block, as a percent. const change24hRaw = (t.stats24h && t.stats24h.priceChange != null) ? t.stats24h.priceChange : (t.stats6h && t.stats6h.priceChange != null) ? t.stats6h.priceChange : (t.stats1h && t.stats1h.priceChange != null) ? t.stats1h.priceChange : null; const change24h = change24hRaw != null ? Number(change24hRaw) : null; const organicScore = Number(t.organicScore); return { mint: t.id || t.mint || t.address || "", symbol: t.symbol || "", name: t.name || "", liquidity: isFinite(liquidity) ? liquidity : null, volume24h: (volume24h != null && isFinite(volume24h)) ? volume24h : null, mcap: isFinite(mcap) ? mcap : null, price: isFinite(price) ? price : null, change24h: (change24h != null && isFinite(change24h)) ? change24h : null, holderCount: isFinite(Number(t.holderCount)) ? Number(t.holderCount) : null, organicScore: isFinite(organicScore) ? organicScore : null, isVerified: t.isVerified === true, }; } // ----------------------------------------------------------------------------- // Direct Jupiter fetch: richest-token match for a subject (cached + deduped) // ----------------------------------------------------------------------------- // At scale, many keywords resolve to the same subject ("bonk", "buy bonk", // "bonk solana"). Without caching, each page re-hits Jupiter's search endpoint, // which is slow and invites rate limiting on a large build. Two protections: // - TTL cache: a resolved subject is reused for JUP_CACHE_TTL ms. // - In-flight dedupe: concurrent identical queries share ONE promise, so a // burst of pages for the same token fires a single network call. const _richTokenCache = new Map(); // key -> { ts, value } const _richTokenInflight = new Map(); // key -> Promise function _cacheSet(key, value) { _richTokenCache.set(key, { ts: Date.now(), value }); // Bound memory: when over cap, drop the oldest ~10% by insertion order. if (_richTokenCache.size > JUP_CACHE_MAX) { const drop = Math.ceil(JUP_CACHE_MAX * 0.1); let i = 0; for (const k of _richTokenCache.keys()) { _richTokenCache.delete(k); if (++i >= drop) break; } } } // Disambiguate same-symbol tokens. Liquidity is the primary signal — the // deepest pool is almost always the real asset — but a verified token with a // strong organic score should beat an unverified one with marginally more // liquidity (liquidity can be faked; organic score is built to resist that). function rankToken(m) { const liq = m.liquidity || 0; const verifiedBoost = m.isVerified ? 1.5 : 1.0; const organicBoost = 1 + ((m.organicScore || 0) / 100); // 1.0 - 2.0 return liq * verifiedBoost * organicBoost; } async function _fetchRichestUncached(query) { const url = `${JUP_TOKENS_SEARCH}?query=${encodeURIComponent(query)}`; const res = await fetchWithTimeout(url, {}, JUP_TIMEOUT_MS); if (!res.ok) { console.log(`[VERIXIA-LIVE] Jupiter search HTTP ${res.status} for "${query}"`); return null; } const data = await res.json(); const tokens = Array.isArray(data) ? data : (data.tokens || []); if (!tokens.length) return null; const metrics = tokens.map(readTokenMetrics).filter(m => m.mint); if (!metrics.length) return null; metrics.sort((a, b) => rankToken(b) - rankToken(a)); return metrics[0]; } // Returns metrics for the best-match token for `query`, or null. Never throws. async function fetchRichestTokenDirect(query) { if (!query) return null; const key = String(query).toLowerCase().trim(); const cached = _richTokenCache.get(key); if (cached && (Date.now() - cached.ts) < JUP_CACHE_TTL) return cached.value; const inflight = _richTokenInflight.get(key); if (inflight) return inflight; const p = (async () => { try { return await _fetchRichestUncached(query); } catch (err) { console.log(`[VERIXIA-LIVE] Jupiter search failed for "${query}": ${err.message}`); return null; // transient failure -> no live data this page, no fallback text } })(); _richTokenInflight.set(key, p); try { const value = await p; _cacheSet(key, value); // cache nulls too: avoids re-hammering a dead query within TTL return value; } finally { _richTokenInflight.delete(key); } } // Exposed so a long-running build can clear live-data state between runs, // mirroring the engine's resetOutputGate / resetAntiRepetition pattern. function resetLiveDataCache() { _richTokenCache.clear(); _richTokenInflight.clear(); } // Turn a metrics object into one dense, number-rich observation line. function metricsToObservation(m) { if (!m) return null; const sym = m.symbol || "token"; const parts = []; const price = fmtPriceUsd(m.price); if (price) parts.push(price); const pct = m.change24h != null ? fmtPct(m.change24h) : null; if (pct) parts.push(`${pct} 24h`); const liq = m.liquidity != null ? fmtUsd(m.liquidity) : null; if (liq) parts.push(`${liq} liquidity`); const vol = m.volume24h != null ? fmtUsd(m.volume24h) : null; if (vol) parts.push(`${vol} 24h volume`); const mc = m.mcap != null ? fmtUsd(m.mcap) : null; if (mc) parts.push(`${mc} market cap`); const holders = m.holderCount != null && m.holderCount > 0 ? `${m.holderCount.toLocaleString("en-US")} holders` : null; if (holders) parts.push(holders); if (parts.length < 2) return null; // not enough real data to be worth weaving const tag = m.isVerified ? "verified, deepest pool" : "deepest pool"; return `Live ${sym} on Solana via Jupiter (${tag}): ${parts.join(", ")}.`; } // Build the enriched observation set: engine's own observations PLUS the // wrapper's direct richest-token pull. Deduped, never throws. Also returns the // resolved token (with its mint) so the caller can surface it — this is the // mint that makes any later cross-source join (e.g. CoinGecko by contract // address) safe, which was the whole reason for selecting by liquidity. async function buildEnrichedObservations(keyword, payload) { const out = []; const seen = new Set(); let liveToken = null; const add = (line) => { if (!line) return; const key = line.toLowerCase(); if (seen.has(key)) return; seen.add(key); out.push(line); }; // 1. Recover what the engine already fetched — this is the data the old // wrapper silently dropped. for (const obs of (payload.observations || [])) add(obs); // 2. Direct richest-token pull for the keyword subject. try { const subj = (extractKeywordSubject(keyword) || {}).subject || ""; if (subj && subj.length >= 2 && subj.length <= 12) { const m = await fetchRichestTokenDirect(subj); if (m) { liveToken = m; add(metricsToObservation(m)); } } } catch (e) { console.log(`[VERIXIA-LIVE] enrichment skipped for "${keyword}": ${e.message}`); } return { observations: out, liveToken }; } // ----------------------------------------------------------------------------- // Expansion (single pass) // ----------------------------------------------------------------------------- async function expandBodyOnce(keyword, currentBody, intent, observations, temperature) { if (!process.env.OPENAI_API_KEY) return null; const obsBlock = (observations && observations.length) ? `\n\nLIVE DATA — weave these exact numbers into the body verbatim, at least two of them, naturally in prose:\n` + observations.map(o => `- ${o}`).join("\n") : ""; const system = `You expand SEO body copy for Verixia (a non-custodial swap on Solana — no KYC, no accounts, no limits, Jupiter routing, 400ms blocks, sub-cent fees, brand tokens settled in USDC, Wonderland memes, bridges from 69 chains). EXPAND the provided text to AT LEAST ${TARGET_MIN_WORDS} words and MORE depth — 500 to 700 words is the target. Rules: - Plain prose only. NO bullets, NO numbered lists, NO headers, NO HTML, NO markdown. - 5 to 8 paragraphs separated by blank lines. - Write like a Solana-native trader, not a brokerage. Direct, confident, concrete. - Add depth: specific numbers, concrete scenarios, comparison to CEX/Ethereum, real examples. - When live data is provided, weave the actual figures in verbatim — do not round them away or replace them with vague phrasing. - Keep "Verixia" naturally woven in. Don't repeat the same sentence structure. - DO NOT add: "do your own research", "consult a financial advisor", "not financial advice", "users should", "in conclusion", "overall". - Vary sentence length. Mix short punchy lines with longer ones. - Keep the original ideas but enrich them. Don't just pad. Output only the expanded body. No preamble, no commentary.`; const user = `Keyword: "${keyword}"\nIntent: ${intent}\n\nExpand this body to ${TARGET_MIN_WORDS}+ words with more depth:\n\n${currentBody}${obsBlock}`; try { const res = await fetchWithTimeout(OPENAI_API_URL, { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${process.env.OPENAI_API_KEY}`, }, body: JSON.stringify({ model: OPENAI_MODEL, temperature, max_tokens: EXPANSION_MAX_TOKENS, messages: [ { role: "system", content: system }, { role: "user", content: user }, ], }), }, EXPANSION_TIMEOUT_MS); if (!res.ok) { const txt = await res.text().catch(() => ""); console.log(`[VERIXIA-EXPAND] OpenAI HTTP ${res.status} (temp ${temperature}) for "${keyword}": ${txt.slice(0, 200)}`); return null; } const data = await res.json(); const expanded = data?.choices?.[0]?.message?.content || ""; return expanded.trim() || null; } catch (err) { console.log(`[VERIXIA-EXPAND] failed (temp ${temperature}) for "${keyword}": ${err.message}`); return null; } } // ----------------------------------------------------------------------------- // Expansion (two passes, parallel, scored, take best — always return one) // ----------------------------------------------------------------------------- // Runs both expansions concurrently. Scores each with the engine's own // validateSeoContent against the enriched observations. Keeps the higher score. // Never throws; if both expansions come back empty, returns null so the caller // Count how many concrete live figures from the observations actually survived // into the expanded body. This is the real target: a pass that wove in more // live figures is the better page. We extract only SEMANTICALLY MEANINGFUL // figures, in three classes, to avoid noise: // - currency: $145.47, $90.0M, $2.0M (anchored by the $ sign) // - percent: +3.21%, 42.5% (anchored by the % sign) // - grouped: 2,342,610 (thousands-separated, e.g. holders) // Crucially this does NOT match bare tokens like "24" (from "24h") or "400ms" // (the "ms" no longer gets mis-read as a magnitude unit), which previously // inflated the score equally across both candidates and added noise. function extractNumbers(text) { const s = String(text || ""); const out = new Set(); // Currency: optional space after $, optional B/M/K magnitude (case-insensitive // for the body's sake, normalized to uppercase). for (const m of s.match(/\$\s?\d[\d,]*(?:\.\d+)?\s?[bmkBMK]?/g) || []) { const norm = m.replace(/\s+/g, "").replace(/,/g, "").replace(/^\$/, "").toUpperCase(); if (norm) out.add("$" + norm); } // Percent: optional leading sign, dropped on normalize so "+3.21%" == "3.21%". for (const m of s.match(/[+-]?\d[\d,]*(?:\.\d+)?\s?%/g) || []) { const norm = m.replace(/\s+/g, "").replace(/,/g, "").replace(/^[+-]/, ""); out.add("%" + norm); } // Grouped big numbers (>= 1,000 with separators) — holder counts etc. for (const m of s.match(/\b\d{1,3}(?:,\d{3})+\b/g) || []) { out.add("#" + m.replace(/,/g, "")); } return out; } function liveWeaveScore(body, observations) { if (!observations || !observations.length) return 0; const bodyNums = extractNumbers(body); if (!bodyNums.size) return 0; const obsNums = new Set(); for (const o of observations) { for (const n of extractNumbers(o)) obsNums.add(n); } let hits = 0; for (const n of obsNums) { if (bodyNums.has(n)) hits++; } return hits; } // ----------------------------------------------------------------------------- // Expansion (two passes, parallel, scored, take best — always return one) // ----------------------------------------------------------------------------- // Runs both expansions concurrently. Primary ranking is the live-data weave // count (how many real figures landed in the body) — that is the actual goal, // and it fits the 500-700 word expanded format. The engine's validateSeoContent // is kept as a SECONDARY tiebreaker only: its hard 380-word / 4-paragraph cap // fights the expansion target, so it can't be the primary signal here (and the // big file can't be touched to relax it). Never throws; if both expansions come // back empty, returns null so the caller keeps the engine's original body. // No fallbacks beyond "keep what we had". async function expandBodyBestOfTwo(keyword, currentBody, intent, observations) { if (!process.env.OPENAI_API_KEY) return null; const settled = await Promise.allSettled( EXPANSION_TEMPERATURES.map(t => expandBodyOnce(keyword, currentBody, intent, observations, t) ) ); const candidates = []; settled.forEach((r, i) => { if (r.status === "fulfilled" && r.value && wordCount(r.value) > 0) { const body = r.value; const weave = liveWeaveScore(body, observations); let engineScore = 0; try { engineScore = (validateSeoContent(body, keyword, intent, observations) || {}).score || 0; } catch (e) { engineScore = 0; // scoring failure shouldn't drop a real candidate } candidates.push({ body, weave, engineScore, words: wordCount(body), temp: EXPANSION_TEMPERATURES[i] }); } }); if (!candidates.length) return null; // Primary: more live figures woven in. Tiebreak: engine score, then length. candidates.sort((a, b) => (b.weave - a.weave) || (b.engineScore - a.engineScore) || (b.words - a.words) ); const best = candidates[0]; console.log( `[VERIXIA-EXPAND] keyword="${keyword}" passes=${candidates.length} ` + `ranked=[${candidates.map(c => `${c.temp}:weave${c.weave}/eng${c.engineScore}/${c.words}w`).join(", ")}] ` + `chose temp=${best.temp} weave=${best.weave} words=${best.words}` ); return best.body; } // ----------------------------------------------------------------------------- // Open-mode acceptance: accept almost anything with real content // ----------------------------------------------------------------------------- function acceptOpen(payload) { const body = payload.body || ""; const words = wordCount(body); const paragraphs = splitParagraphs(body); if (!body.trim()) { payload.valid = false; payload.score = 0; payload.reasons = [...(payload.reasons || []), "open-mode: empty body"]; return payload; } if (looksLikeListOrHtml(body)) { payload.valid = false; payload.score = 0; payload.reasons = [...(payload.reasons || []), "open-mode: list/html formatting"]; return payload; } let score = 72; if (words >= TARGET_MIN_WORDS) score += 15; else if (words >= 350) score += 8; else if (words >= 200) score += 0; else score -= 10; if (paragraphs.length >= 4) score += 5; payload.valid = true; payload.score = Math.max(0, Math.min(100, score)); payload.paragraphs = paragraphs; payload.reasons = [...(payload.reasons || []), `open-mode: accepted (${words} words, ${paragraphs.length} paragraphs)`]; return payload; } // ----------------------------------------------------------------------------- // Wrapped generateSeoPagePayload // ----------------------------------------------------------------------------- async function generateSeoPagePayload(keyword) { const payload = await __origGenerateSeoPagePayload(keyword); // Add framingName if missing if (payload && !payload.framingName && payload.intent) { try { payload.framingName = selectFramingForKeyword(keyword, payload.intent); } catch (e) { /* non-fatal */ } } if (!payload || !payload.body) { console.log(`[VERIXIA-OPEN] keyword="${keyword}" engine returned empty body, nothing to expand`); return payload; } const intent = payload.intent || "general_native"; const initialWords = wordCount(payload.body); // Enrich observations: recover the engine's own + direct richest-token pull. const { observations: enrichedObs, liveToken } = await buildEnrichedObservations(keyword, payload); payload.observations = enrichedObs; // surface the fuller set downstream too // Surface the resolved token (mint + metrics). The mint is the safe join key // for any later cross-source enrichment; downstream can also render it. if (liveToken) { payload.liveToken = { mint: liveToken.mint, symbol: liveToken.symbol, name: liveToken.name, price: liveToken.price, change24h: liveToken.change24h, liquidity: liveToken.liquidity, volume24h: liveToken.volume24h, mcap: liveToken.mcap, holderCount: liveToken.holderCount, organicScore: liveToken.organicScore, isVerified: liveToken.isVerified, }; } console.log( `[VERIXIA-OPEN] keyword="${keyword}" engine body=${initialWords} words, ` + `observations=${enrichedObs.length}, ` + `liveToken=${liveToken ? `${liveToken.symbol}:${liveToken.mint.slice(0, 6)}…` : "none"}, ` + `expanding (2 parallel passes)...` ); // Always expand toward 500+ words if we're under target. if (initialWords < TARGET_MIN_WORDS) { const expanded = await expandBodyBestOfTwo(keyword, payload.body, intent, enrichedObs); if (expanded && wordCount(expanded) > initialWords) { payload.body = expanded; payload.paragraphs = splitParagraphs(expanded); console.log(`[VERIXIA-OPEN] keyword="${keyword}" expanded to ${wordCount(expanded)} words`); } else { console.log(`[VERIXIA-OPEN] keyword="${keyword}" expansion produced nothing better, keeping original`); } } acceptOpen(payload); console.log( `[VERIXIA-OPEN] keyword="${keyword}" ` + `intent="${intent}" ` + `framing="${payload.framingName || ""}" ` + `final_words=${wordCount(payload.body)} ` + `paragraphs=${(payload.paragraphs || []).length} ` + `score=${payload.score} valid=${payload.valid}` ); return payload; } // ----------------------------------------------------------------------------- // Re-export everything else unchanged // ----------------------------------------------------------------------------- export { generateSeoPagePayload, detectSeoIntent, detectSubIntent, extractKeywordSubject, validateSeoContent, fetchJupQuote, fetchJupPrices, fetchJupRecent, fetchJupTokenSearch, resetOutputGate, pageJitter, selectFramingForKeyword, getBuildReport, resetAntiRepetition, resetLiveDataCache, };
